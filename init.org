(setup-define :autoload
  (lambda (func)
    (let ((fn (if (memq (car-safe func) '(quote function))
                  (cadr func)
                func)))
      `(unless (fboundp (quote ,fn))
         (autoload (function ,fn) ,(symbol-name (setup-get 'feature)) nil t))))
  :documentation "Autoload COMMAND if not already bound."
  :repeatable t
  :signature '(FUNC ...))

  (defun meow-vim-append ()
    "Like vim, move to the end of selection, switch to INSERT state."
    (interactive)
    (if meow--temp-normal
        (progn
          (message "Quit temporary normal mode")
          (meow--switch-state 'motion))
      (unless (region-active-p)
        (when (and meow-use-cursor-position-hack
                   (< (point) (point-max)))
          (forward-char 1))
        (forward-char 1))
      (when (region-active-p)
        (meow--direction-forward)
        (meow--cancel-selection))
      (meow--switch-state 'insert)))

    "用指定字符包围选中区域"

    (when (use-region-p)

          (goto-char end)
          (insert char)
          (goto-char start)

#+begin_src elisp
    (defun nowis/smart-bind (key condition-fn new-fn &optional keymap)
      "Bind KEY to NEW-FN when CONDITION-FN is true, otherwise use original binding."
      (let* ((map (or keymap global-map))
             (original-binding (lookup-key map (kbd key))))
        (define-key map (kbd key)
                    (lambda ()
                      (interactive)
                      (if (funcall condition-fn)
                          (call-interactively new-fn)
                        (if original-binding
                            (call-interactively original-binding)
                          (message "No original binding for %s" key)))))))

(defun ghostel--buffers ()
  (match-buffers '(derived-mode . ghostel-mode)))

(defun ghostel--window ()
  (get-window-with-predicate
   (lambda (w) (buffer-match-p '(derived-mode . ghostel-mode) (window-buffer w)))))

(defun ghostel--ago (time)
  (if time (format "%ds ago" (floor (float-time (time-subtract nil time)))) "-"))

(defun ghostel--annotate (name)
  (when-let* ((buf (get-buffer (substring-no-properties name))))
    (with-current-buffer buf
      (concat "  "
              (propertize (abbreviate-file-name default-directory)
                          'face 'font-lock-comment-face)
              (propertize (format "  active %s" (ghostel--ago ghostel--last-output-time))
                          'face 'shadow)))))

(defvar ghostel-consult-source
  (list :name "ghostel" :narrow ?g :category 'ghostel
        :face 'consult-buffer :history 'buffer-name-history
        :annotate #'ghostel--annotate :state #'consult--buffer-state
        :new (lambda (name)
               (let ((ghostel-buffer-name (format "*%s*" (string-trim name "*"))))
                 (ghostel)))
        :items (lambda () (mapcar #'buffer-name (ghostel--buffers)))))

  "Toggle ghostel; C-u to pick via consult."
  (interactive "P")
  (cond
   (choose-buffer (consult-buffer (list ghostel-consult-source)))
   ((ghostel--window) (delete-window (ghostel--window)))
   ((car (ghostel--buffers)) (pop-to-buffer (car (ghostel--buffers))))

;;; ─── 分组（基于 Category）────────────────────────
(defun nowis/agenda-entry-olpath (&optional marker)
  "获取当前 agenda 条目的项目路径。优先 ARCHIVE_OLPATH，否则回退到 CATEGORY。"
  (if marker
      (with-current-buffer (marker-buffer marker)
        (save-excursion
          (goto-char marker)
          (or (org-entry-get (point) "ARCHIVE_OLPATH")
              (org-get-category (point)))))
    (let ((m (get-text-property (point) 'org-marker)))
      (if m
          (or (org-entry-get m "ARCHIVE_OLPATH")
              (get-text-property (point) 'org-category))
        (get-text-property (point) 'org-category)))))

(defun nowis/agenda-add-category-separators ()
  "在不同 olpath 分组间插入分隔标题行。需在排序完成后运行。"
  (save-excursion
    (goto-char (point-min))
    (let ((last-group nil))
      (while (not (eobp))
        (when-let* ((group (nowis/agenda-entry-olpath)))
          (unless (equal group last-group)
            (let ((inhibit-read-only t))
              (insert (propertize (concat "◆ " group "\n")
                                  'face 'org-agenda-structure)))
            (setq last-group group)))
        (forward-line)))))

(add-hook 'org-agenda-finalize-hook #'nowis/agenda-add-category-separators)

(defun nowis/center-header (text)
  "将 TEXT 在当前 agenda 窗口中居中。"
  (let* ((width (window-width (get-buffer-window)))
         (pad (max 0 (/ (- width (string-width text)) 2))))
    (concat (make-string pad ?\s) text)))

;;; ─── 近期完成（agenda 已统一含 archive，直接复用全集）─────────────────
(defun nowis/agenda-closed-date ()
  "返回当前 agenda 条目的 CLOSED 日期 MM-DD（用于 prefix 显示）。"
  (let ((c (org-entry-get (point) "CLOSED")))
    (if c (format-time-string "%m-%d" (org-time-string-to-time c)) "     ")))

(defun nowis/agenda--closed-secs (marker)
  "MARKER 处条目的 CLOSED 时间戳（秒），无则 0。"
  (with-current-buffer (marker-buffer marker)
    (save-excursion
      (goto-char marker)
      (let ((c (org-entry-get (point) "CLOSED")))
        (if c (float-time (org-time-string-to-time c)) 0)))))

(defun nowis/agenda-cmp-closed (a b)
  "先按项目聚拢、组内按 CLOSED 倒序比较两行。配合 user-defined-down（取反）。"
  (let* ((ma (get-text-property 0 'org-marker a))
         (mb (get-text-property 0 'org-marker b))
         (ca (if ma (nowis/agenda-entry-olpath ma) ""))
         (cb (if mb (nowis/agenda-entry-olpath mb) "")))
    (cond
     ((not (string= ca cb)) (if (string> ca cb) 1 -1))
     (t (let ((sa (if ma (nowis/agenda--closed-secs ma) 0))
              (sb (if mb (nowis/agenda--closed-secs mb) 0)))
          (cond ((> sa sb) 1) ((< sa sb) -1) (t nil)))))))

                ((org-agenda-overriding-header ,(format "近 %d 天完成" days))
                 (org-agenda-prefix-format "  %(nowis/agenda-closed-date)  ")
                 (org-agenda-cmp-user-defined #'nowis/agenda-cmp-closed)
                 (org-agenda-sorting-strategy '(user-defined-down)))))))
